# agent.py
class AIAgent:
    def __init__(self, api_key: str, model: str = "gpt-4"):
        self.api_key = api_key
        self.model = model
        self.messages = []
    
    def chat(self, user_message: str) -> str:
        self.messages.append({"role": "user", "content": user_message})
        
        response = openai.ChatCompletion.create(
            model=self.model,
            messages=self.messages
        )
        
        assistant_message = response.choices[0].message
        self.messages.append(assistant_message)
        
        return assistant_message["content"]

# cli.py
def main():
    agent = AIAgent(api_key=os.getenv("OPENAI_API_KEY"))
    while True:
        user_input = input("> ")
        if user_input.lower() in ("exit", "quit"):
            break
        response = agent.chat(user_input)
        print(response)

if __name__ == "__main__":
    main()
```

# registry.py
class ToolRegistry:
    def __init__(self):
        self._tools = {}
    
    def register(self, name, schema, handler):
        self._tools[name] = {"schema": schema, "handler": handler}
    
    def get_tool(self, name):
        return self._tools.get(name)

# agent.py - 添加工具循环
def run_with_tools(self):
    while True:
        response = self.call_llm()
        
        if response.tool_calls:
            for call in response.tool_calls:
                tool = self.registry.get_tool(call.function.name)
                result = tool["handler"](**json.loads(call.function.arguments))
                self.messages.append({
                    "role": "tool",
                    "tool_call_id": call.id,
                    "content": result
                })
        else:
            return response.content
```

## 17.3 缺失的 Agent OS 能力

| 能力 | 状态 | 说明 |
|-----|------|-----|
| **进程隔离** | 部分 | Docker/SSH 后端可隔离，但非强制 |
| **虚拟内存** | 无 | 无真正的内存隔离 |
| **安全沙箱** | 基础 | 仅 terminal 后端有隔离 |
| **进程间通信** | 基础 | 仅通过工具调用 |
| **系统调用** | 基础 | 工具调用类似系统调用 |
| **设备驱动** | 基础 | environments 后端类似驱动 |
| **文件系统权限** | 基础 | path_security 有限检查 |
