## 13.2 新增 Tool

```python
# Step 1: 创建工具文件
# tools/my_tool.py

from tools.registry import registry
import json

def my_tool(param: str, task_id: str = None) -> str:
    """我的自定义工具"""
    return json.dumps({"result": f"Processed: {param}"})

registry.register(
    name="my_tool",
    toolset="custom",
    schema={
        "name": "my_tool",
        "description": "我的自定义工具",
        "parameters": {
            "type": "object",
            "properties": {
                "param": {"type": "string"}
            }
        }
    },
    handler=lambda args, **kw: my_tool(**kw),
)

# Step 2: 添加到工具集
# toolsets.py
TOOLSETS["custom"] = {
    "tools": ["my_tool", ...],
}
```
